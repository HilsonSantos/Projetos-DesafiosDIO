IF OBJECT_ID ('DBO.CARGA_CLIENTES', 'U') IS NULL
BEGIN
    CREATE TABLE CARGA_CLIENTES (
        ID_CLIENTE INT NOT NULL,
        NOME VARCHAR(15) NOT NULL,
        CIDADE VARCHAR(20) NOT NULL,

END
